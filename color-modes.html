BOOTSTRAP COLOR MODES

▶ Introduction to Color Modes

Bootstrap 5.3+ introduces built-in support for color modes, enabling you to easily implement light and dark themes in your applications. Color modes use CSS custom properties (variables) to dynamically switch between different color schemes.

This feature allows users to choose their preferred theme (light, dark, or auto) and provides a consistent experience across your application.

Pro Tip: Color modes work by toggling the data-bs-theme attribute on the <html> element, which switches CSS custom property values.

▶ Implementing Color Modes

Enable Color Modes in Sass

Color modes are enabled by default in Bootstrap 5.3+. Ensure you're using the correct version and import structure.

// main.scss
@import "bootstrap/scss/functions";
@import "bootstrap/scss/variables";
@import "bootstrap/scss/variables-dark"; // Dark mode variables
@import "bootstrap/scss/maps";
@import "bootstrap/scss/mixins";
@import "bootstrap/scss/root"; // CSS custom properties
// ... rest of imports

Set Default Theme

Set the default color mode by adding the data-bs-theme attribute to your HTML element.

<!DOCTYPE html>
<html lang="en" data-bs-theme="dark">
  <head>
    <!-- Meta tags -->
    <title>My App</title>
    <link href="bootstrap.css" rel="stylesheet">
  </head>
  <body>
    <!-- Your content -->
  </body>
</html>

Create Theme Toggle

Implement a toggle button to switch between light and dark modes using JavaScript.

<button class="btn btn-primary" onclick="toggleTheme()">
  <i class="fas fa-moon"></i> Toggle Theme
</button>

<script>
  function toggleTheme() {
    const html = document.documentElement;
    const currentTheme = html.getAttribute('data-bs-theme');
    const newTheme = currentTheme === 'dark' ? 'light' : 'dark';
    
    html.setAttribute('data-bs-theme', newTheme);
    
    // Optional: Save preference to localStorage
    localStorage.setItem('bs-theme', newTheme);
  }
  
  // Optional: Load saved theme on page load
  const savedTheme = localStorage.getItem('bs-theme');
  if (savedTheme) {
    document.documentElement.setAttribute('data-bs-theme', savedTheme);
  }
</script>

▶ Customizing Color Modes

Custom Theme Variables

Create custom color modes by extending the default Sass variables:

// _variables.scss
// Light mode defaults
$primary: #7C3AED;
$secondary: #06B6D4;

// Dark mode overrides
$primary-dark: #8B5CF6;
$secondary-dark: #22D3EE;

// main.scss
@import "bootstrap/scss/functions";

// Light mode variables
$primary: $primary;
$secondary: $secondary;
@import "bootstrap/scss/variables";

// Dark mode variables
$primary: $primary-dark;
$secondary: $secondary-dark;
@import "bootstrap/scss/variables-dark";

// Reset to light mode for maps
$primary: $primary-light;
$secondary: $secondary-light;
@import "bootstrap/scss/maps";
@import "bootstrap/scss/mixins";
@import "bootstrap/scss/root";

Component-specific Themes

Apply different themes to specific components using the data-bs-theme attribute:

<!-- Global dark theme -->
<html data-bs-theme="dark">
  <body>
    <!-- Light themed section -->
    <section data-bs-theme="light">
      <div class="card">
        <div class="card-body">
          This card uses light theme
        </div>
      </div>
    </section>
    
    <!-- Dark themed section (inherits from parent) -->
    <section>
      <div class="card">
        <div class="card-body">
          This card uses dark theme
        </div>
      </div>
    </section>
  </body>
</html>

▶ Advanced Color Mode Features

Auto Mode with Media Query

Implement automatic theme switching based on system preferences:

<script>
  // Check system preference on load
  function initTheme() {
    const systemPrefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
    const savedTheme = localStorage.getItem('bs-theme');
    
    if (savedTheme === 'auto' || (!savedTheme && systemPrefersDark)) {
      document.documentElement.setAttribute('data-bs-theme', 'dark');
    } else if (savedTheme) {
      document.documentElement.setAttribute('data-bs-theme', savedTheme);
    }
  }
  
  // Listen for system preference changes
  window.matchMedia('(prefers-color-scheme: dark)').addEventListener('change', e => {
    const savedTheme = localStorage.getItem('bs-theme');
    if (savedTheme === 'auto') {
      document.documentElement.setAttribute('data-bs-theme', e.matches ? 'dark' : 'light');
    }
  });
  
  // Initialize on load
  initTheme();
</script>

Multiple Custom Themes

Create additional custom themes beyond light and dark:

// Create custom theme variables
:root[data-bs-theme="blue"] {
  --bs-body-bg: #1e3a8a;
  --bs-body-color: #dbeafe;
  --bs-primary: #3b82f6;
  --bs-secondary: #60a5fa;
  --bs-success: #10b981;
  --bs-border-color: #374151;
}

:root[data-bs-theme="green"] {
  --bs-body-bg: #064e3b;
  --bs-body-color: #d1fae5;
  --bs-primary: #10b981;
  --bs-secondary: #34d399;
  --bs-success: #059669;
  --bs-border-color: #065f46;
}

// JavaScript to switch between multiple themes
function setTheme(themeName) {
  document.documentElement.setAttribute('data-bs-theme', themeName);
  localStorage.setItem('bs-theme', themeName);
}

▶ Complete Color Mode Implementation

Full Implementation Example

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Color Modes Example</title>
  <link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.3/dist/css/bootstrap.min.css" rel="stylesheet">
  <link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/bootstrap-icons@1.11.1/font/bootstrap-icons.css">
</head>
<body>
  <nav class="navbar navbar-expand-lg">
    <div class="container">
      <a class="navbar-brand" href="#">My App</a>
      <div class="dropdown">
        <button class="btn btn-outline-secondary dropdown-toggle" type="button" data-bs-toggle="dropdown">
          <i class="bi bi-palette"></i> Theme
        </button>
        <ul class="dropdown-menu">
          <li><button class="dropdown-item" onclick="setTheme('light')">Light</button></li>
          <li><button class="dropdown-item" onclick="setTheme('dark')">Dark</button></li>
          <li><button class="dropdown-item" onclick="setTheme('auto')">Auto</button></li>
        </ul>
      </div>
    </div>
  </nav>

  <main class="container my-5">
    <div class="row">
      <div class="col-md-6">
        <div class="card">
          <div class="card-body">
            <h5 class="card-title">Card Title</h5>
            <p class="card-text">This card adapts to the current theme.</p>
            <button class="btn btn-primary">Primary Button</button>
          </div>
        </div>
      </div>
      <div class="col-md-6" data-bs-theme="light">
        <div class="card">
          <div class="card-body">
            <h5 class="card-title">Always Light</h5>
            <p class="card-text">This section is always in light mode.</p>
            <button class="btn btn-primary">Primary Button</button>
          </div>
        </div>
      </div>
    </div>
  </main>

  <script src="https://cdn.jsdelivr.net/npm/bootstrap@5.3.3/dist/js/bootstrap.bundle.min.js"></script>
  <script>
    function setTheme(theme) {
      const html = document.documentElement;
      
      if (theme === 'auto') {
        const systemPrefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
        html.setAttribute('data-bs-theme', systemPrefersDark ? 'dark' : 'light');
      } else {
        html.setAttribute('data-bs-theme', theme);
      }
      
      localStorage.setItem('bs-theme', theme);
    }

    // Initialize theme
    const savedTheme = localStorage.getItem('bs-theme') || 'auto';
    setTheme(savedTheme);

    // Listen for system preference changes
    window.matchMedia('(prefers-color-scheme: dark)').addEventListener('change', e => {
      if (localStorage.getItem('bs-theme') === 'auto') {
        setTheme('auto');
      }
    });
  </script>
</body>
</html>
This complete example shows a fully functional color mode implementation with theme persistence, system preference detection, and component-specific theming.

Color Mode Benefits

  • User Preference: Allow users to choose their preferred viewing experience.
  • Accessibility: Improve readability and reduce eye strain in different lighting conditions.
  • Battery Saving: Dark mode can save battery on OLED screens.
  • Modern UX: Provide a contemporary user experience with theme switching.
